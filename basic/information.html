# 内核版本
$ uname -r
5.15.0-48-generic

$ echo $(. /etc/os-release;echo $ID $VERSION_ID)
ubuntu 22.04


# 永久修改
$ sudo echo "PermanentHostname" > /etc/hostname

# 临时修改
$ sudo hostname "TemporaryHostname"
